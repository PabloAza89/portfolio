        <div>
          <MuiSwitch
            onClick={handleSetEnableLockRotate}
            checked={ enableLockRotate ? true : false }
          />
          Rotate
        </div>

        <div>
          Dont forget to 'lock' the padlock to changes take effect.
        </div>

      </div>

    </div>
  )
